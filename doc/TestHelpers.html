def test_log_methods
  # Clear adaptibrew ouput from brewer
  @brewer.clear

  clear_log(@log)
  assert_true(File.zero?(@log))

  # Get some output
  @brewer.script('python_tester')
  assert_true(@brewer.out.any?)

  # Write output
  write_log(@log, @brewer.out)
  assert_true(!File.zero?(@log))
end
